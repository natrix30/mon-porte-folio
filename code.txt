import pygame
import random

class Carte:
    def __init__(self, valeur, couleur):
        self.valeur = valeur
        self.couleur = couleur
        self.nb_passes = 0

    def code(self):
        valeurs_map = {11: "J",12: "Q",13: "K",14: "A"}
        v = valeurs_map.get(self.valeur, str(self.valeur))
        return self.couleur + v

    def __repr__(self):
        return self.code()

class Paquet:
    def __init__(self):
        couleurs = ['H', 'D', 'C', 'S']
        valeurs = list(range(2, 15))
        self.pile = []
        for valeur in valeurs:
            for couleur in couleurs:
                self.pile.append(Carte(valeur, couleur))

    def distribuer(self, nb_joueurs):
        random.shuffle(self.pile)
        joueurs = [[] for _ in range(nb_joueurs)]
        while self.pile:
            for j in joueurs:
                if self.pile:
                    j.append(self.pile.pop())
        for main in joueurs:
            main.sort(key=lambda c: (1 if c.valeur == 14 else c.valeur))
        return joueurs

class Jeu:
    def __init__(self, nb_joueurs):
        paquet = Paquet()
        self.joueurs = paquet.distribuer(nb_joueurs)
        self.nb_joueurs = nb_joueurs
        self.joueur_actuel = 0
        self.valeur_attendue = 14
        self.nb_passes = 0

    def carte_jouable(self, carte):
        return carte.valeur == self.valeur_attendue

    def jouer_carte(self, joueur_index, carte_index):
        carte = self.joueurs[joueur_index][carte_index]
        if not self.carte_jouable(carte):
            return None
        self.joueurs[joueur_index].pop(carte_index)
        if self.valeur_attendue == 14:
            self.valeur_attendue = 2
        elif self.valeur_attendue == 13:
            self.valeur_attendue = 14
        else:
            self.valeur_attendue += 1
        self.nb_passes = 0
        self.joueur_actuel = (self.joueur_actuel + 1) % self.nb_joueurs
        return carte

    def verifier_gain_special(self, carte):
        mapping = {('H', 13): 0,('S', 12): 1,('D', 10): 2,('C', 11): 3,('D', 7): 4}
        return mapping.get((carte.couleur, carte.valeur), None)

    def passer(self):
        self.nb_passes += 1
        if self.nb_passes >= self.nb_joueurs:
            if self.valeur_attendue == 14:
                self.valeur_attendue = 2
            elif self.valeur_attendue == 13:
                self.valeur_attendue = 14
            else:
                self.valeur_attendue += 1
            print("Nouvelle valeur demandée :", self.valeur_attendue)
            self.nb_passes = 0
        self.joueur_actuel = (self.joueur_actuel + 1) % self.nb_joueurs

def load_cards():
    cards = {}
    couleurs = ['H', 'D', 'C', 'S']
    valeurs = list(range(2, 15))
    for valeur in valeurs:
        for couleur in couleurs:
            temp = Carte(valeur, couleur)
            name = temp.code()  # H2, D2 etc
            path = f"cards/{name}.png"
            image = pygame.image.load(path)
            image = pygame.transform.scale(image, (CARD_WIDTH, CARD_HEIGHT))
            cards[name] = image
    return cards
